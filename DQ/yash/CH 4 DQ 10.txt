->SELECT FIRST_NAME , SALARY , HIRE_DATE ,
CASE WHEN TO_CHAR(HIRE_DATE,'MON') = 'JUN' OR TO_CHAR(HIRE_DATE,'YY') = 98 THEN SALARY + 1500
     WHEN TO_CHAR(HIRE_DATE,'MON') = 'JAN' OR TO_CHAR(HIRE_DATE,'YY') = 99 THEN SALARY + 2000
     WHEN TO_CHAR(HIRE_DATE,'MON') = 'JUL' OR TO_CHAR(HIRE_DATE,'YY') = ANY(90,92,94) THEN SALARY + 1000
ELSE SALARY END "NEW SALARY" FROM EMPLOYEES ORDER BY HIRE_DATE;
