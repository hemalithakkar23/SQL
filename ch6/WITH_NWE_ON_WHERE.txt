1. SELECT H.LAST_NAME , H.HIRE_DATE , E.LAST_NAME , H.HIRE_DATE
FROM EMPLOYEES H JOIN EMPLOYEES E
ON E.HIRE_DATE < H.HIRE_DATE
WHERE H.LAST_NAME = 'Hartstein';
2.SELECT G.LAST_NAME , G.COMMISSION_PCT , H.LAST_NAME , H.COMMISSION_PCT
FROM EMPLOYEES G
JOIN EMPLOYEES H
ON (H.COMMISSION_PCT > G.COMMISSION_PCT )
WHERE G.LAST_NAME = 'Grant';

3.SELECT A.LAST_NAME , A.SALARY + (A.COMMISSION_PCT*A.SALARY) A_COMM_PCT ,
 E.LAST_NAME , E.SALARY + (E.COMMISSION_PCT*E.SALARY) E_COMM_PCT
FROM EMPLOYEES A JOIN EMPLOYEES E
ON (E.SALARY + (E.COMMISSION_PCT*E.SALARY) < A.SALARY + (A.COMMISSION_PCT*A.SALARY)
)WHERE A.LAST_NAME = 'Abel';

4.SELECT H.LAST_NAME, TO_CHAR (H.HIRE_DATE , 'Q'),
 E.LAST_NAME, TO_CHAR (E.HIRE_DATE , 'Q')
FROM EMPLOYEES H JOIN EMPLOYEES E
ON (TO_CHAR (H.HIRE_DATE , 'Q')) = TO_CHAR (E.HIRE_DATE , 'Q')
WHERE H.LAST_NAME = 'Higgins';
