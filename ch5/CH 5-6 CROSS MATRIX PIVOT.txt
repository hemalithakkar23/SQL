1 DISPLAY COUNT OF EMPLOYEES FOR DEPARTMENT_ID DEPT90 , DEPT50 DEPT60 (CROSS TAB REPORT)
SELECT COUNT(DECODE(DEPARTMENT_ID,90,'*')) DEPT_90,
        COUNT(DECODE(DEPARTMENT_ID,50,'*')) DEPT_50,
        COUNT(DECODE(DEPARTMENT_ID,60,'*')) DEPT_60
FROM EMPLOYEES;

SELECT COUNT(CASE DEPARTMENT_ID WHEN 90 THEN '*' END) DEPT_90 ,
        COUNT(CASE DEPARTMENT_ID WHEN 50 THEN '*' END ) DEPT_50  ,
        COUNT(CASE DEPARTMENT_ID WHEN 60 THEN '*' END) DEPT_60 
FROM EMPLOYEES;

2 DISPLAY LOWEST SALARY OF EMPLOYEES FOR JOB_ID IT_PROG , ST_CLERK , SA_REP , AD_VP (PIVOTE REPORT)
SELECT MIN(DECODE(JOB_ID,'IT_PROG',SALARY)) IT_PROG_MIN_SAL,
                MIN(DECODE(JOB_ID,'ST_CLERK',SALARY)) ST_CLERK_MIN_SAL,
                MIN(DECODE(JOB_ID,'SA_REP',SALARY)) SA_REP_MIN_SAL,
                MIN(DECODE(JOB_ID,'AD_VP',SALARY)) AD_VP_MIN_SAL
FROM EMPLOYEES;

SELECT MIN(CASE WHEN JOB_ID='IT_PROG' THEN SALARY END) IT_PROG_MIN_SAL,
        MIN(CASE WHEN JOB_ID='SA_REP' THEN SALARY END) SA_REP_MIN_SAL,
        MIN(CASE WHEN JOB_ID='AD_VP' THEN SALARY END) AD_VP_MIN_SAL,
        MIN(CASE WHEN JOB_ID='ST_CLERK' THEN SALARY END) ST_CLERK_MIN_SAL
FROM EMPLOYEES;

3 DISPLAY HIGHEST SALARY OF EMPLOYEES FOR MANAGER_ID 100 , 124 , 149 (CROSS TAB REPORT)

16 DISPLAY W1_SUM_SAL   W2_MAX_SAL   W3_AVG_SAL  W4_MIN_SAL  W5_COUNT (W1 = WEEK1, ...)
SELECT SUM(DECODE(TO_CHAR(HIRE_DATE,'W'),'1',SALARY)) W1_SUM_SAL,
        MAX(CASE WHEN TO_CHAR(HIRE_DATE,'W')= '2' THEN SALARY END) W2_MAX_SAL,
        AVG(DECODE(TO_CHAR(HIRE_DATE,'W'),'3',SALARY)) W3_AVG_SAL,
        MIN(CASE WHEN TO_CHAR(HIRE_DATE,'W')='4' THEN SALARY END) W4_MIN_SAL,
        COUNT(CASE TO_CHAR(HIRE_DATE,'W') WHEN '5' THEN SALARY END) W5_COUNT
        FROM EMPLOYEES;

        SUM(CASE WHEN TO_CHAR(HIRE_DATE,'Q')='2' THEN SALARY END ) Q2_SUM_SAL  , 
        SUM(CASE WHEN TO_CHAR(HIRE_DATE,'Q')='3' THEN SALARY END ) Q3_SUM_SAL  ,

        FROM EMPLOYEES
