1 DISPLAY COUNT OF EMPLOYEES FOR DEPARTMENT_ID DEPT90 , DEPT50 DEPT60 (CROSS TAB REPORT)
-> SELECT COUNT(CASE WHEN DEPARTMENT_ID = 90 THEN '*' END) DEPT90 ,
       COUNT(CASE WHEN DEPARTMENT_ID = 50 THEN '*' END) DEPT50 ,
       COUNT(CASE WHEN DEPARTMENT_ID = 60 THEN '*' END) DEPT60 FROM EMPLOYEES WHERE DEPARTMENT_ID IN (90,50,60);
       
2 DISPLAY LOWEST SALARY OF EMPLOYEES FOR JOB_ID IT_PROG , ST_CLERK , SA_REP , AD_VP (PIVOTE REPORT)
->SELECT MIN(SALARY) , JOB_ID FROM EMPLOYEES WHERE JOB_ID IN ('IT_PROG','ST_CLERK','SA_REP','AD_VP') GROUP BY JOB_ID;

3 DISPLAY HIGHEST SALARY OF EMPLOYEES FOR MANAGER_ID 100 , 124 , 149 (CROSS TAB REPORT)
->SELECT MAX(SALARY) , MANAGER_ID FROM EMPLOYEES WHERE MANAGER_ID IN (100,124,149) GROUP BY MANAGER_ID;

4 DISPLAY DEPARTMENT_ID WISE AVERAGE SALARY FOR IT_PROG , SA_REP , AD_VP , ST_CLERK (MATRIX REPORT)
->SELECT DEPARTMENT_ID , JOB_ID ,AVG(SALARY) FROM EMPLOYEES WHERE JOB_ID = ANY('IT_PROG','SA_REP','AD_VP','ST_CLERK') GROUP BY DEPARTMENT_ID , JOB_ID;

5 DISPLAY MANAGER_ID WISE TOTAL OF SALARY FOR DEPARTMENT_ID 50,60,80,90 (PIVOTE REPORT)
->SELECT MANAGER_ID , DEPARTMENT_ID , SUM(SALARY) FROM EMPLOYEES WHERE DEPARTMENT_ID IN (50,60,80,90) GROUP BY MANAGER_ID , DEPARTMENT_ID;

6 DISPLAY DEPARTMENT_ID WISE TOTAL OF SALARY FOR IT_PROG , SA_RAP , AD_VP , ST_CLERK ONLY FOR DEPARTMENT_ID 90,50,80,60 (MATRIX REPORT)
->SELECT DEPARTMENT_ID , JOB_ID , SUM(SALARY) FROM EMPLOYEES WHERE  JOB_ID IN ('IT_PROG','SA_RAP','AD_VP','ST_CLERK') GROUP BY JOB_ID , DEPARTMENT_ID;

7 DISPLAY LOWEST SALARY FOR DEPARTMENT 20, AVERAGE SALARY FOR DEPARTMENT 50, HIGHEST SALARY FOR DEPARTMENT 90 (USING DECODE)
->SELECT MIN(CASE WHEN DEPARTMENT_ID = 20 THEN SALARY END) MIN_DEPT_20 ,
         AVG(CASE WHEN DEPARTMENT_ID = 50 THEN SALARY END) AVG_DEPT_50 ,
         MAX(CASE WHEN DEPARTMENT_ID = 90 THEN SALARY END) MAX_DEPT_90 FROM EMPLOYEES;
 
8 DISPLAY LOWEST SALARY FOR MANAGER 100, AVERAGE SALARY FOR MANAGER 124, HIGHEST SALARY FOR MANAGER 149 (USING CASE 1st SYNTAX)
->SELECT MIN(CASE WHEN MANAGER_ID = 100 THEN SALARY END) MIN_MGID_100 ,
         AVG(CASE WHEN MANAGER_ID = 124 THEN SALARY END) AVG_MGID_124 ,
         MAX(CASE WHEN MANAGER_ID = 149 THEN SALARY END) MAX_MGID_149 FROM EMPLOYEES;

9 DISPLAY LOWEST SALARY FOR IT_PROG, AVERAGE SALARY FOR ST_CLERK, HIGHEST SALARY FOR SA_REP (USING CASE 2nd SYNTAX)
->SELECT MIN(CASE WHEN JOB_ID IN ('IT_PROG') THEN SALARY END) MIN_JOBID_IT_PROG ,
         AVG(CASE WHEN JOB_ID IN ('ST_CLERK') THEN SALARY END) AVG_JOBID_ST_CLERK ,
         MAX(CASE WHEN JOB_ID IN ('SA_REP') THEN SALARY END) AVG_JOBID_SA_REP FROM EMPLOYEES;

10 DISPLAY MANAGER_ID WISE COUNT OF EMPLOYEES FOR IT_PROG (DECODE), COUNT OF EMPLOYEES FOR MANAGER 100 (CASE)
-> SELECT MANAGER_ID , JOB_ID , COUNT(DECODE(JOB_ID , 'IT_PROG' , SALARY )) FROM EMPLOYEES GROUP BY MANAGER_ID,JOB_ID;
-> SELECT MANAGER_ID , COUNT(CASE WHEN MANAGER_ID = 100 THEN SALARY END) FROM EMPLOYEES GROUP BY MANAGER_ID;

11 DISPLAY LOWEST SALARY FOR SA_REP (DECODE), AVERAGE SALARY FOR MANAGER 149 (CASE I), HIGHEST SALARY FOR DEPARTMENT_ID 60 (CASE II)
->SELECT MIN(DECODE(JOB_ID,'SA_REP',SALARY)) MIN_JOBID_SA_REP ,
         AVG(CASE WHEN MANAGER_ID = 149 THEN SALARY END) AVG_MGID_149 ,
         MAX(CASE WHEN DEPARTMENT_ID = 60 THEN SALARY END) MAX_DPID_60 FROM EMPLOYEES;

12 DISPLAY COUNT OF EMPLOYEES HIRED IN 1995 (DECODE), COUNT OF EMPLOYEES HIRED IN 1997 (CASE I), COUNT OF EMPLOYEES HIRED IN 1999 (CASE II)
->SELECT COUNT(DECODE(TO_CHAR(HIRE_DATE,'RRRR'),'1995','*')) C_HD_95 ,
       COUNT(CASE WHEN TO_CHAR(HIRE_DATE,'RRRR') IN  '1997' THEN '*' END) C_HD_97 ,
       COUNT(CASE WHEN TO_CHAR(HIRE_DATE,'RRRR') IN  '1999' THEN '*' END) C_HD_99 FROM EMPLOYEES;

13 DISPLAY DEPARTMENT_ID WISE COUNT OF EMPLOYEES HIRED IN 1995 (DECODE), COUNT OF EMPLOYEES HIRED IN 1997 (CASE I), COUNT OF EMPLOYEES HIRED IN 1999 (CASE II)
-> SELECT DEPARTMENT_ID , COUNT(DECODE(TO_CHAR(HIRE_DATE,'RRRR'),'1995','*')) C_HD_95 ,
        COUNT(CASE WHEN TO_CHAR(HIRE_DATE,'RRRR') IN  '1997' THEN '*' END) C_HD_97 ,
        COUNT(CASE WHEN TO_CHAR(HIRE_DATE,'RRRR') IN  '1999' THEN '*' END) C_HD_99 FROM EMPLOYEES GROUP BY DEPARTMENT_ID;

14 DISPLAY HIRE YEAR WISE COUNT OF EMPLOYEES FOR DEPT 50, COUNT OF EMPLOYEES FOR MANAGER 100 AND COUNT OF EMLOYEES FOR IT_PROG.
->SELECT TO_CHAR(HIRE_DATE,'YY') , COUNT(CASE WHEN DEPARTMENT_ID = 50 THEN '*' END) C_DEPT_50 ,
                                 COUNT(CASE WHEN MANAGER_ID = 100 THEN '*' END) C_MGID_100 ,
                                 COUNT(CASE WHEN JOB_ID = 'IT_PROG' THEN '*' END) C_JOB_ID_IT_PROG FROM EMPLOYEES GROUP BY HIRE_DATE;

15 DISPLAY Q1_MIN_SAL   Q2_AVG_SAL   Q3_MAX_SAL   Q4_SUM_SAL   TOTAL_SALARY
->SELECT MIN(CASE WHEN TO_CHAR(HIRE_DATE,'Q') = 1 THEN SALARY END) Q1_MIN_SAL ,
       AVG(CASE WHEN TO_CHAR(HIRE_DATE,'Q') = 2 THEN SALARY END) Q2_AVG_SAL ,
       MAX(CASE WHEN TO_CHAR(HIRE_DATE,'Q') = 3 THEN SALARY END) Q3_MAX_SAL ,
       SUM(CASE WHEN TO_CHAR(HIRE_DATE,'Q') = 4 THEN SALARY END) Q4_SUM_SAL ,
       SUM(SALARY) FROM EMPLOYEES;

16 DISPLAY W1_SUM_SAL   W2_MAX_SAL   W3_AVG_SAL  W4_MIN_SAL  W5_COUNT (W1 = WEEK1, ...)
->SELECT SUM(CASE WHEN TO_CHAR(HIRE_DATE,'W') = 1 THEN SALARY END) W1_SUM_SAL ,
       MAX(CASE WHEN TO_CHAR(HIRE_DATE,'W') = 2 THEN SALARY END) W2_MAX_SAL ,
       AVG(CASE WHEN TO_CHAR(HIRE_DATE,'W') = 3 THEN SALARY END) W3_AVG_SAL ,
       MIN(CASE WHEN TO_CHAR(HIRE_DATE,'W') = 4 THEN SALARY END) W4_MIN_SAL ,
       COUNT(CASE WHEN TO_CHAR(HIRE_DATE,'W') = 5 THEN '*' END) W5_COUNT FROM EMPLOYEES;

17 DISPLAY COUNT OF FIRST_NAME WITH ZERO 'e' , 1 OCCURRENCE OF 'e' , 2 OCCURRENCES OF 'e' : CNT_0_E  CNT_1_E  CNT_2_E
