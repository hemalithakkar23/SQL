CREATE TABLE PERSONAL_INFO(
NAME VARCHAR(40),
BIRTH_DATE DATE ,
WEIGHT NUMBER(6,3),
PHOTOGRAPH BFILE ) ;

CREATE TABLE FIXED_DEPOSIT (
BANK_NAME VARCHAR(50),
INVESTMENT_DATE TIMESTAMP WITH TIME ZONE ,
YM_PERIOD INTERVAL YEAR TO MONTH ,
DAYS_PERIOD INTERVAL DAY TO SECOND );

CREATE TABLE EUROPE_TOUR (
DESTINATION VARCHAR(20),
REACH_DATE TIMESTAMP TIME ZONE ,
DAYS_PRIOD INTERVAL DAY TO SECOND );

CREATE TABLE DISCOVERY_SERIALS ( 
SERIAL_NAME VARCHAR(200),
SHOW_NO NUMBER ,
DURATION INTERVAL YEAR TO MONTH ,
VIDEO_FILE BFILE ) ;


CREATE TABLE BOOK_LIBRARY(
BOOK_NAME VARCHAR(40),
AUTHOR VARCHAR(60),
NO_OF_PAGES NUMBER ,
BOOK_PDF BFILE);

CREATE TABLE MNC(
CUSTOMER_NAME VARCHAR(70),
CONTANCT_NO VARCHAR(20),
MOBILE_NO VARCHAR(20),
BILL_AMOUNT NUMBER(10,2));

DROP TABLE  PERSONAL_INFO ;
DROP TABLE  FIXED_DEPOSIT;
DROP TABLE    EUROPE_TOUR ;
DROP TABLE  DISCOVERY_SERIALS;
DROP TABLE  BOOK_LIBRARY;
DROP TABLE   MNC;
