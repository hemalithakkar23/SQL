REM CREATE TABLE E1 AS SELECT * FROM EMPLOYEES;

1 DELETE THOSE WHO REPORTS TO MANAGER 124
DELETE FROM E1 WHERE MANAGER_ID = 124 ;

2 DELETE THOSE WHO WERE HIRED IN YEAR 99
DELETE FROM E1 WHERE TO_CHAR(HIRE_DATE , 'RRRR') = '1999';

3 DELETE THOSE WHO WORKS IN Seattle CITY
DELETE E1 WHERE DEPARTMENT_ID IN
 (SELECT DEPARTMENT_ID FROM DEPARTMENTS WHERE LOCATION_ID IN  (SELECT LOCATION_ID FROM LOCATIONS WHERE UPPER(CITY)='SEATTLE'));

4 DELETE THOSE WHO REPORTS TO Steven
DELETE E1 WHERE MANAGER_ID IN (SELECT EMPLOYEE_ID FROM EMPLOYEES WHERE UPPER(FIRST_NAME) = 'STEVEN');

5 DELETE ALL EXCEPT THOSE WHO GETS SALARY IN THE RANGE OF 6000 TO 11000
DELETE E1 WHERE SALARY NOT BETWEEN 6000 AND 11000;

6 DELETE THOSE WHO WERE HIRED IN 3rd WEEK OF ANY MONTH.
DELETE FROM E1 WHERE TO_CHAR(HIRE_DATE , 'W') ='3';

7 DELETE THOSE WHO WERE HIRED UPTO 9th DATE
DELETE FROM E1 WHERE TO_CHAR(HIRE_DATE ,'DD') <=9;

8 DELETE THOSE WHO GETS SALARY IN 'C' GRADE
DELETE FROM E1 WHERE
 SALARY BETWEEN (SELECT  LOWEST_SAL FROM JOB_GRADES WHERE GRADE_LEVEL = 'C') AND (SELECT  HIGHEST_SAL FROM JOB_GRADES WHERE GRADE_LEVEL = 'C');

9 DELETE THOSE WHO REPORTS TO Kevin
DELETE E1 WHERE MANAGER_ID IN (SELECT EMPLOYEE_ID FROM EMPLOYEES WHERE UPPER(FIRST_NAME) = 'KEVIN');

10 DELETE THOSE HAVE GOT PROMOTION
DELETE FROM E1 WHERE EMPLOYEE_ID IN (SELECT EMPLOYEE_ID FROM JOB_HISTORY);
